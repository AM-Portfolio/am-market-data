  --namespace monitoring \
  --create-namespace \
  --set grafana.enabled=true \
  --set prometheus.enabled=true \
  --set prometheus.alertmanager.persistentVolume.enabled=false \
  --set prometheus.server.persistentVolume.enabled=true \
  --set prometheus.server.persistentVolume.size=5Gi \
  --set loki.persistence.enabled=true \
  --set loki.persistence.size=5Gi \
  --set grafana.persistence.enabled=true \
  --set grafana.persistence.size=1Gi \
  --set grafana.resources.requests.cpu=50m \
  --set grafana.resources.requests.memory=64Mi \
  --set grafana.resources.limits.cpu=100m \
  --set grafana.resources.limits.memory=128Mi \
  --set prometheus.server.resources.requests.cpu=50m \
  --set prometheus.server.resources.requests.memory=128Mi \
  --set prometheus.server.resources.limits.cpu=100m \
  --set prometheus.server.resources.limits.memory=256Mi \
  --set loki.resources.requests.cpu=50m \
  --set loki.resources.requests.memory=64Mi \
  --set loki.resources.limits.cpu=100m \
  --set loki.resources.limits.memory=128Mi \
